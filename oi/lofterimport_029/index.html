constintmaxn = 100009;

qry q[maxm];

 

inlineboolcmpQry(constqry& a, constqry& b) {

    returnw[a. l] < w[b. l] || (w[a. l] == w[b. l] && a. r < b. r);

inlinevoidchg(intp, intv) {

intqry(intl, intr) {

    ints = 0;

        for(inti = r; w[i] == w[r]; -- i)

            if(c[i])

                ++ s;

    }

    returns;

}
